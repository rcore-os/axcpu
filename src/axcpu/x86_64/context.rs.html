86/// A 512-byte memory region for the FXSAVE/FXRSTOR instruction to save and
87/// restore the x87 FPU, MMX, XMM, and MXCSR registers.
88///
89/// See <https://www.felixcloutier.com/x86/fxsave> for more details.
90#[allow(missing_docs)]
91#[repr(C, align(16))]
92#[derive(Debug)]
93pub struct FxsaveArea {
94    pub fcw: u16,
95    pub fsw: u16,
96    pub ftw: u16,
97    pub fop: u16,
98    pub fip: u64,
99    pub fdp: u64,
100    pub mxcsr: u32,
101    pub mxcsr_mask: u32,
102    pub st: [u64; 16],
103    pub xmm: [u64; 32],
104    _padding: [u64; 12],
105}
106
107static_assertions::const_assert_eq!(core::mem::size_of::<FxsaveArea>(), 512);

109/// Extended state of a task, such as FP/SIMD states.
110pub struct ExtendedState {
111    /// Memory region for the FXSAVE/FXRSTOR instruction.
112    pub fxsave_area: FxsaveArea,
113}
114
115#[cfg(feature = "fp-simd")]
116impl ExtendedState {
117    #[inline]
118    fn save(&mut self) {
119        unsafe { core::arch::x86_64::_fxsave64(&mut self.fxsave_area as *mut _ as *mut u8) }
120    }
121
122    #[inline]
123    fn restore(&self) {
124        unsafe { core::arch::x86_64::_fxrstor64(&self.fxsave_area as *const _ as *const u8) }
125    }
126
127    const fn default() -> Self {
128        let mut area: FxsaveArea = unsafe { core::mem::MaybeUninit::zeroed().assume_init() };
129        area.fcw = 0x37f;
130        area.ftw = 0xffff;
131        area.mxcsr = 0x1f80;
132        Self { fxsave_area: area }
133    }
134}
135
136impl fmt::Debug for ExtendedState {
137    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
138        f.debug_struct("ExtendedState")
139            .field("fxsave_area", &self.fxsave_area)
140            .finish()
141    }
142}

144/// Saved hardware states of a task.
145///
146/// The context usually includes:
147///
148/// - Callee-saved registers
149/// - Stack pointer register
150/// - Thread pointer register (for thread-local storage, currently unsupported)
151/// - FP/SIMD registers
152///
153/// On context switch, current task saves its context from CPU to memory,
154/// and the next task restores its context from memory to CPU.
155///
156/// On x86_64, callee-saved registers are saved to the kernel stack by the
157/// `PUSH` instruction. So that [`rsp`] is the `RSP` after callee-saved
158/// registers are pushed, and [`kstack_top`] is the top of the kernel stack
159/// (`RSP` before any push).
160///
161/// [`rsp`]: TaskContext::rsp
162/// [`kstack_top`]: TaskContext::kstack_top
163#[derive(Debug)]
164pub struct TaskContext {
165    /// The kernel stack top of the task.
166    pub kstack_top: VirtAddr,
167    /// `RSP` after all callee-saved registers are pushed.
168    pub rsp: u64,
169    /// Thread Local Storage (TLS).
170    pub fs_base: usize,
171    /// The `gs_base` register value.
172    #[cfg(feature = "uspace")]
173    pub gs_base: usize,
174    /// Extended states, i.e., FP/SIMD states.
175    #[cfg(feature = "fp-simd")]
176    pub ext_state: ExtendedState,
177    /// The `CR3` register value, i.e., the page table root.
178    #[cfg(feature = "uspace")]
179    pub cr3: memory_addr::PhysAddr,
180}
181
182impl TaskContext {
183    /// Creates a dummy context for a new task.
184    ///
185    /// Note the context is not initialized, it will be filled by [`switch_to`]
186    /// (for initial tasks) and [`init`] (for regular tasks) methods.
187    ///
188    /// [`init`]: TaskContext::init
189    /// [`switch_to`]: TaskContext::switch_to
190    pub fn new() -> Self {
191        Self {
192            kstack_top: va!(0),
193            rsp: 0,
194            fs_base: 0,
195            #[cfg(feature = "uspace")]
196            cr3: crate::asm::read_kernel_page_table(),
197            #[cfg(feature = "fp-simd")]
198            ext_state: ExtendedState::default(),
199            #[cfg(feature = "uspace")]
200            gs_base: 0,
201        }
202    }
203
204    /// Initializes the context for a new task, with the given entry point and
205    /// kernel stack.
206    pub fn init(&mut self, entry: usize, kstack_top: VirtAddr, tls_area: VirtAddr) {
207        unsafe {
208            // x86_64 calling convention: the stack must be 16-byte aligned before
209            // calling a function. That means when entering a new task (`ret` in `context_switch`
210            // is executed), (stack pointer + 8) should be 16-byte aligned.
211            let frame_ptr = (kstack_top.as_mut_ptr() as *mut u64).sub(1);
212            let frame_ptr = (frame_ptr as *mut ContextSwitchFrame).sub(1);
213            core::ptr::write(
214                frame_ptr,
215                ContextSwitchFrame {
216                    rip: entry as _,
217                    ..Default::default()
218                },
219            );
220            self.rsp = frame_ptr as u64;
221        }
222        self.kstack_top = kstack_top;
223        self.fs_base = tls_area.as_usize();
224    }
225
226    /// Changes the page table root in this context.
227    ///
228    /// The hardware register for page table root (`CR3` for x86) will be
229    /// updated to the next task's after [`Self::switch_to`].
230    #[cfg(feature = "uspace")]
231    pub fn set_page_table_root(&mut self, cr3: memory_addr::PhysAddr) {
232        self.cr3 = cr3;
233    }
234
235    /// Switches to another task.
236    ///
237    /// It first saves the current task's context from CPU to this place, and then
238    /// restores the next task's context from `next_ctx` to CPU.
239    pub fn switch_to(&mut self, next_ctx: &Self) {
240        #[cfg(feature = "fp-simd")]
241        {
242            self.ext_state.save();
243            next_ctx.ext_state.restore();
244        }
245        #[cfg(any(feature = "tls", feature = "uspace"))]
246        unsafe {
247            self.fs_base = crate::asm::read_thread_pointer();
248            crate::asm::write_thread_pointer(next_ctx.fs_base);
249        }
250        #[cfg(feature = "uspace")]
251        unsafe {
252            // Switch gs base for user space.
253            self.gs_base = x86::msr::rdmsr(x86::msr::IA32_KERNEL_GSBASE) as usize;
254            x86::msr::wrmsr(x86::msr::IA32_KERNEL_GSBASE, next_ctx.gs_base as u64);
255            super::gdt::write_tss_rsp0(next_ctx.kstack_top);
256            if next_ctx.cr3 != self.cr3 {
257                crate::asm::write_user_page_table(next_ctx.cr3);
258                // writing to CR3 has flushed the TLB
259            }
260        }
261        unsafe { context_switch(&mut self.rsp, &next_ctx.rsp) }
262    }
263}

265#[unsafe(naked)]
266unsafe extern "C" fn context_switch(_current_stack: &mut u64, _next_stack: &u64) {
267    naked_asm!(
268        "
269        .code64
270        push    rbp
271        push    rbx
272        push    r12
273        push    r13
274        push    r14
275        push    r15
276        mov     [rdi], rsp
277
278        mov     rsp, [rsi]
279        pop     r15
280        pop     r14
281        pop     r13
282        pop     r12
283        pop     rbx
284        pop     rbp
285        ret",
286    )
