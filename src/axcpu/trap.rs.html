18/// A slice of syscall handler functions.
19#[cfg(feature = "uspace")]
20#[doc(cfg(feature = "uspace"))]
21#[def_trap_handler]
22pub static SYSCALL: [fn(&TrapFrame, usize) -> isize];
23
24#[allow(unused_macros)]
25macro_rules! handle_trap {
26    ($trap:ident, $($args:tt)*) => {{
27        let mut iter = $crate::trap::$trap.iter();
28        if let Some(func) = iter.next() {
29            if iter.next().is_some() {
30                warn!("Multiple handlers for trap {} are not currently supported", stringify!($trap));
31            }
32            func($($args)*)
33        } else {
34            warn!("No registered handler for trap {}", stringify!($trap));
35            false
36        }
37    }}
38}
39
40/// Call the external syscall handler.
41#[cfg(feature = "uspace")]
42pub(crate) fn handle_syscall(tf: &TrapFrame, syscall_num: usize) -> isize {
43    SYSCALL[0](tf, syscall_num)
44}
